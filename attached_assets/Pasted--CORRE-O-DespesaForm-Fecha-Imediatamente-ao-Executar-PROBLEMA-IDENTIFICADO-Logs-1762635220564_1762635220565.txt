# 🔧 CORREÇÃO - DespesaForm Fecha Imediatamente ao Executar

## 🎯 PROBLEMA IDENTIFICADO

**Logs do Console:**
```
▶️ ExecucaoOrcamentaria: Executando despesa planejada
✅ Abrindo DespesaForm para EXECUTAR (pré-preenchido)
🔍 DEBUG: Carregando Despesa para Edição
[FECHA IMEDIATAMENTE] ❌
```

**Causa:** DespesaForm não reconhece `modoExecucao` e está tratando como edição normal, mas ao detectar que é uma despesa PLANEJADA (status="PLANEJADA"), está bloqueando ou fechando automaticamente.

---

## ✅ SOLUÇÃO

### **Arquivo:** `src/components/DespesaForm.jsx`

**Localizar a assinatura do componente:**
```javascript

  despesaParaEditar,
  emendaId,
  somenteLeitura,
  emendaInfo,
  onClose,
  onSave,
}) => {
```

**SUBSTITUIR POR:**
```javascript

  emendaId,
```

```javascript
const handleSalvar = async () => {
  // ... validações
```

```javascript
    if (isExecucao) {
      // ✅ MODO EXECUÇÃO: Deletar planejada + Criar executada
      console.log("▶️ Executando transição: PLANEJADA → EXECUTADA");

      // ✅ CRIAÇÃO NORMAL

    console.error("❌ Erro ao salvar despesa:", error);
```

## 📋 CÓDIGO COMPLETO DO DESPESAFORM (INÍCIO)

```javascript
import React, { useState, useEffect } from 'react';
import { 
  collection, 
  addDoc, 
  updateDoc, 
  deleteDoc, 
  doc 
} from 'firebase/firestore';
import { db } from '../firebase/firebaseConfig';
import { formatarMoedaInput, parseValorMonetario } from '../utils/formatters';

const DespesaForm = ({
  despesaParaEditar,
  emendaId,
  somenteLeitura = false,
  modoExecucao = false, // ✅ NOVA PROP
  emendaInfo,
  onClose,
  onSave,
}) => {
  // ✅ DETECTAR MODO
  const isExecucao = modoExecucao === true;
  const isEdicao = !isExecucao && despesaParaEditar?.id;
  
  console.log("🔍 DespesaForm - Modo:", {
    modoExecucao,
    isExecucao,
    isEdicao,
    despesaId: despesaParaEditar?.id,
    status: despesaParaEditar?.status
  });
  
  const [formData, setFormData] = useState({
    emendaId: emendaId || '',
    discriminacao: '',
    fornecedor: '',
    valor: '',
    numeroEmpenho: '',
    numeroNota: '',
    cnpj: '',
    // ... outros campos
  });
  
  const [salvando, setSalvando] = useState(false);
  
  // ✅ CARREGAR DADOS DA DESPESA
  useEffect(() => {
    if (despesaParaEditar) {
      if (isExecucao) {
        console.log("▶️ Carregando PLANEJADA para EXECUÇÃO");
        console.log("Dados da planejada:", despesaParaEditar);
      } else {
        console.log("✏️ Carregando para EDIÇÃO");
      }
      
      setFormData(prev => ({
        ...prev,
        emendaId: despesaParaEditar.emendaId || emendaId,
        naturezaDespesa: despesaParaEditar.estrategia || despesaParaEditar.naturezaDespesa || '',
        valor: formatarMoedaInput(despesaParaEditar.valor?.toString() || '0'),
        
        // ✅ DISCRIMINAÇÃO
        discriminacao: isExecucao ? '' : (despesaParaEditar.discriminacao || ''),
        
        // ✅ OUTROS CAMPOS (apenas em edição)
        ...(isEdicao && {
          fornecedor: despesaParaEditar.fornecedor || '',
          numeroEmpenho: despesaParaEditar.numeroEmpenho || '',
          numeroNota: despesaParaEditar.numeroNota || '',
          cnpj: despesaParaEditar.cnpj || '',
          // ... outros campos de edição
        })
      }));
    }
  }, [despesaParaEditar, isExecucao, isEdicao, emendaId]);
  
  // ✅ BLOQUEAR FECHAMENTO AUTOMÁTICO
  useEffect(() => {
    if (isExecucao && despesaParaEditar) {
      console.log("⚠️ Modo execução - formulário permanece aberto");
    }
  }, [isExecucao, despesaParaEditar]);
  
  // ✅ SALVAMENTO
  const handleSalvar = async (e) => {
    e?.preventDefault();
    
    if (salvando) {
      console.log("⏳ Salvamento já em andamento, ignorando clique");
      return;
    }
    
    // Validações...
    
    setSalvando(true);
    
    try {
      const dadosParaSalvar = {
        emendaId: formData.emendaId,
        discriminacao: formData.discriminacao,
        fornecedor: formData.fornecedor,
        valor: parseValorMonetario(formData.valor),
        numeroEmpenho: formData.numeroEmpenho,
        numeroNota: formData.numeroNota,
        cnpj: formData.cnpj,
        naturezaDespesa: formData.naturezaDespesa,
        // ... outros campos
      };
      
      if (isExecucao) {
        // ✅ TRANSIÇÃO: PLANEJADA → EXECUTADA
        console.log("▶️ Executando transição PLANEJADA → EXECUTADA");
        
        // 1. Criar executada
        const dadosExecutada = {
          ...dadosParaSalvar,
          status: "EXECUTADA",
          criadaEm: new Date().toISOString(),
          executadaEm: new Date().toISOString(),
        };
        
        await addDoc(collection(db, "despesas"), dadosExecutada);
        console.log("✅ Despesa executada criada");
        
        // 2. Deletar planejada
        if (despesaParaEditar?.id) {
          await deleteDoc(doc(db, "despesas", despesaParaEditar.id));
          console.log("✅ Despesa planejada deletada");
        }
        
      } else if (isEdicao) {
        // ✅ EDIÇÃO
        console.log("✏️ Atualizando despesa existente");
        await updateDoc(doc(db, "despesas", despesaParaEditar.id), dadosParaSalvar);
        
      } else {
        // ✅ CRIAÇÃO
        console.log("🆕 Criando nova despesa");
        await addDoc(collection(db, "despesas"), {
          ...dadosParaSalvar,
          status: "EXECUTADA",
          criadaEm: new Date().toISOString(),
        });
      }
      
      console.log("✅ Salvamento concluído com sucesso");
      onSave?.(); // Callback
      
    } catch (error) {
      console.error("❌ Erro ao salvar:", error);
      alert(`Erro ao salvar despesa: ${error.message}`);
    } finally {
      setSalvando(false);
    }
  };
  
  return (
    <div>
      <h2>
        {somenteLeitura ? "👁️ Visualizar Despesa" : 
         isExecucao ? "▶️ Executar Despesa Planejada" :
         isEdicao ? "✏️ Editar Despesa" : 
         "🆕 Nova Despesa"}
      </h2>
      
      {/* Formulário com 4 seções */}
      <form onSubmit={handleSalvar}>
        {/* Seção 1: Dados Básicos */}
        {/* Seção 2: Empenho/Nota */}
        {/* Seção 3: Datas */}
        {/* Seção 4: Classificação */}
        
        <div>
          <button type="button" onClick={onClose}>
            Cancelar
          </button>
          <button type="submit" disabled={salvando || somenteLeitura}>
            {salvando ? "Salvando..." : 
             isExecucao ? "✅ Confirmar Execução" : 
             isEdicao ? "Atualizar" : "Salvar"}
          </button>
        </div>
      </form>
    </div>
  );
};

export default DespesaForm;
```
