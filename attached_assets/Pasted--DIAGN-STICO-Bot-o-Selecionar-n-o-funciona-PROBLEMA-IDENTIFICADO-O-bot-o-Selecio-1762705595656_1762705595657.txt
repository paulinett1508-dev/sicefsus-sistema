## 🔧 FUNÇÃO handleSelecionarEmenda

**Deve estar ANTES do return:**

```javascript
const handleSelecionarEmenda = (emenda) => {
  console.log("🎯 Emenda selecionada:", emenda);
  
  // 1. Definir emenda selecionada
  setEmendaSelecionada(emenda);
  
  // 2. Calcular valores
  const valorCalculado = emenda.despesas?.reduce(
    (total, desp) => total + (parseFloat(desp.valor) || 0),
    0
  ) || 0;
  
  const valorBanco = parseFloat(emenda.valorExecutado) || 0;
  const diferenca = Math.abs(valorBanco - valorCalculado);
  
  // 3. Definir preview
  setPreview({
    numero: emenda.numeroEmenda || emenda.numero,
    municipio: emenda.municipio,
    uf: emenda.uf,
    valorTotal: parseFloat(emenda.valorTotal) || 0,
    valorBanco: valorBanco,
    valorCalculado: valorCalculado,
    diferenca: diferenca,
    percentualBanco: valorBanco > 0 ? (valorBanco / parseFloat(emenda.valorTotal)) * 100 : 0,
    percentualCalculado: valorCalculado > 0 ? (valorCalculado / parseFloat(emenda.valorTotal)) * 100 : 0,
  });
  
  // 4. Abrir modal
  setMostrarModal(true);
  
  console.log("✅ Modal aberto com preview:", preview);
};
```

## 🔧 FUNÇÃO handleAplicarRecalculo

```javascript
const handleAplicarRecalculo = async () => {
  if (!emendaSelecionada || !preview) {
    alert("❌ Erro: Nenhuma emenda selecionada");
    return;
  }

  if (!window.confirm(
    `Confirma a aplicação do recálculo?\n\n` +
    `Emenda: ${preview.numero}\n` +
    `Valor Atual: ${formatarMoeda(preview.valorBanco)}\n` +
    `Novo Valor: ${formatarMoeda(preview.valorCalculado)}`
  )) {
    return;
  }

  setProcessando(true);

  try {
    const emendaRef = doc(db, "emendas", emendaSelecionada.id);
    
    const novoSaldo = preview.valorTotal - preview.valorCalculado;
    
    await updateDoc(emendaRef, {
      valorExecutado: preview.valorCalculado,
      saldoDisponivel: novoSaldo,
      percentualExecutado: preview.percentualCalculado,
      recalculadoEm: new Date().toISOString(),
      recalculadoPor: "SuperAdmin",
    });

    alert(`✅ Recálculo aplicado com sucesso!\n\nEmenda: ${preview.numero}\nNovo valor executado: ${formatarMoeda(preview.valorCalculado)}`);
    
    // Fechar modal e recarregar
    setMostrarModal(false);
    setEmendaSelecionada(null);
    setPreview(null);
    
    // Recarregar emendas
    handleBuscar();
    
  } catch (error) {
    console.error("❌ Erro ao aplicar recálculo:", error);
    alert(`❌ Erro ao aplicar recálculo: ${error.message}`);
  } finally {
    setProcessando(false);
  }
};
```
