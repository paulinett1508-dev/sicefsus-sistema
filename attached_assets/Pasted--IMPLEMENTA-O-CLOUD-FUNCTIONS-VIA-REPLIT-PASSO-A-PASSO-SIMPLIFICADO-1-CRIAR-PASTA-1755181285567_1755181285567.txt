const functions = require('firebase-functions');
const admin = require('firebase-admin');

// Inicializar com credenciais do projeto
admin.initializeApp();

// Função para excluir usuário completamente
exports.deleteUserComplete = functions.https.onCall(async (data, context) => {
  try {
    // Verificar autenticação
    if (!context.auth) {
      throw new functions.https.HttpsError('unauthenticated', 'Não autenticado');
    }

    // Verificar se é admin
    const callerDoc = await admin.firestore()
      .collection('usuarios')
      .doc(context.auth.uid)
      .get();
    
    if (!callerDoc.exists || callerDoc.data().tipo !== 'admin') {
      throw new functions.https.HttpsError('permission-denied', 'Apenas admins');
    }

    const { uid, firestoreId } = data;

    let results = { firestore: false, auth: false };

    // Excluir do Firestore
    try {
      await admin.firestore().collection('usuarios').doc(firestoreId).delete();
      results.firestore = true;
    } catch (error) {
      console.error('Erro Firestore:', error);
    }

    // Excluir do Auth
    try {
      await admin.auth().deleteUser(uid);
      results.auth = true;
    } catch (error) {
      if (error.code === 'auth/user-not-found') {
        results.auth = true; // Já não existe
      }
      console.error('Erro Auth:', error);
    }

    return {
      success: results.firestore || results.auth,
      details: results
    };

  } catch (error) {
    throw new functions.https.HttpsError('internal', error.message);
  }
});

// 8️⃣ ATUALIZAR userService.js:
import { httpsCallable } from 'firebase/functions';
import { functions } from '../firebase/firebaseConfig';

export const deleteUserComplete = async (firestoreId, authUid) => {
  const deleteFunction = httpsCallable(functions, 'deleteUserComplete');
  const result = await deleteFunction({
    uid: authUid || firestoreId,
    firestoreId: firestoreId
  });
  return result.data;
};
