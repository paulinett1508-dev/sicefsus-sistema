// 2. Adicione o estado salvando:
const [salvando, setSalvando] = useState(false);

// 3. Atualize o handleSubmit completo:
const handleSubmit = async (e) => {
  e.preventDefault();
  
  // Prevenir duplo clique
  if (salvando) return;
  
  // Validar formulário
  if (!validarFormulario()) {
    return;
  }
  
  setSalvando(true); // Ativa estado de salvando
  
  try {
    const dadosEmenda = {
      ...formData,
      valor: parseFloat(formData.valor) || 0,
      saldo: emendaParaEditar 
        ? parseFloat(formData.saldo) || 0 
        : parseFloat(formData.valor) || 0,
      dataAprovacao: formData.dataAprovacao || new Date().toISOString(),
      criadoEm: emendaParaEditar?.criadoEm || new Date(),
      atualizadoEm: new Date(),
      status: 'ativa',
    };
    
    if (emendaParaEditar) {
      // Atualizar emenda existente
      const emendaRef = doc(db, "emendas", emendaParaEditar.id);
      await updateDoc(emendaRef, dadosEmenda);
      
      // Aguardar para mostrar feedback
      await new Promise(resolve => setTimeout(resolve, 800));
      
      setToast({
        show: true,
        message: "✅ Emenda atualizada com sucesso!",
        type: "success",
      });
    } else {
      // Criar nova emenda
      await addDoc(collection(db, "emendas"), dadosEmenda);
      
      // Aguardar para mostrar feedback
      await new Promise(resolve => setTimeout(resolve, 800));
      
      setToast({
        show: true,
        message: "✅ Emenda criada com sucesso!",
        type: "success",
      });
    }
    
    // Aguardar mais um pouco para o usuário ver a mensagem
    await new Promise(resolve => setTimeout(resolve, 500));
    
    // Chamar callback de sucesso
    if (onSuccess) {
      onSuccess();
    }
    
    // Limpar formulário se não for edição
    if (!emendaParaEditar) {
      setFormData({
        numero: '',
        ano: new Date().getFullYear().toString(),
        parlamentar: '',
        valor: '',
        objeto: '',
        municipio: '',
        uf: '',
        tipo: '',
        dataAprovacao: new Date().toISOString().split('T')[0],
        observacoes: '',
      });
    }
    
  } catch (error) {
    console.error("❌ Erro ao salvar emenda:", error);
    let mensagemErro = "❌ Erro ao salvar emenda. ";
    
    if (error.code === 'permission-denied') {
      mensagemErro += "Você não tem permissão para esta operação.";
    } else if (error.code === 'already-exists') {
      mensagemErro += "Já existe uma emenda com este número.";
    } else {
      mensagemErro += "Tente novamente.";
    }
    
    setToast({
      show: true,
      message: mensagemErro,
      type: "error",
    });
  } finally {
    setSalvando(false); // Desativa estado de salvando
  }
};

// 4. Atualize o componente EmendaFormActions:
// Em src/components/emenda/EmendaForm/components/EmendaFormActions.jsx

const EmendaFormActions = ({ 
  onCancel, 
  onSubmit, 
  isEdit = false,
  salvando = false // Receber prop salvando
}) => {
  return (
    <div style={styles.actionsContainer}>
      <button
        type="button"
        onClick={onCancel}
        style={styles.cancelButton}
        disabled={salvando}
      >
        Cancelar
      </button>
      <button
        type="submit"
        onClick={onSubmit}
        style={{
          ...styles.submitButton,
          opacity: salvando ? 0.6 : 1,
          cursor: salvando ? 'not-allowed' : 'pointer',
          transform: salvando ? 'scale(0.98)' : 'scale(1)',
        }}
        disabled={salvando}
      >
        {salvando ? (
          "Processando..."
        ) : (
          isEdit ? 'Atualizar Emenda' : 'Salvar Emenda'
        )}
      </button>
    </div>
  );
};

// 5. No JSX do EmendaForm, passe a prop salvando:

  salvando={salvando} // Passar estado salvando
/>
